        // --- PRECISION ZODIAC LOGIC ---
        function getZodiacIndex(month, day) {
            const mmdd = (month * 100) + day;
            if (mmdd >= 321 && mmdd <= 419) return 0;  // Aries
            if (mmdd >= 420 && mmdd <= 520) return 1;  // Taurus
            if (mmdd >= 521 && mmdd <= 620) return 2;  // Gemini
            if (mmdd >= 621 && mmdd <= 722) return 3;  // Cancer
            if (mmdd >= 723 && mmdd <= 822) return 4;  // Leo
            if (mmdd >= 823 && mmdd <= 922) return 5;  // Virgo
            if (mmdd >= 923 && mmdd <= 1022) return 6; // Libra
            if (mmdd >= 1023 && mmdd <= 1121) return 7; // Scorpio
            if (mmdd >= 1122 && mmdd <= 1221) return 8; // Sag
            if (mmdd >= 1222 || mmdd <= 119) return 9;  // Capricorn
            if (mmdd >= 120 && mmdd <= 218) return 10; // Aquarius
            if (mmdd >= 219 && mmdd <= 320) return 11; // Pisces
            return 0;
        }

        function checkSync() {

        // --- DRAWING FUNCTIONS ---
        function drawWesternWheel(idx) {
            const svg = document.getElementById('westernSVG');
            svg.innerHTML = '<circle cx="200" cy="200" r="180" fill="none" stroke="var(--border)" stroke-width="2" />';
            for(let i=0; i<12; i++) {
                let a = i * 30 * Math.PI/180;
                svg.innerHTML += `<line x1="200" y1="200" x2="${200+180*Math.cos(a)}" y2="${200+180*Math.sin(a)}" stroke="var(--border)" stroke-width="1" />`;
            }
            let sunA = (idx * 30 + 15 + 180) * Math.PI/180; // Added 180 to align visually
            svg.innerHTML += `<circle cx="${200+140*Math.cos(sunA)}" cy="${200+140*Math.sin(sunA)}" r="12" fill="var(--gold)" filter="drop-shadow(0 0 10px var(--gold))" />`;
        }

        function calculateLifePath(dob) {
            let s = dob.replace(/-/g, '').split('').reduce((a,b)=>a+parseInt(b), 0);
            while(s > 9 && s!==11 && s!==22) s = s.toString().split('').reduce((a,b)=>a+parseInt(b), 0);
            return s;
        }
